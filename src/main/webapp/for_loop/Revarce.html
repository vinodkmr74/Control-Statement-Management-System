import java.util.*;
public class Revarce {
public static void main(String[] args) {
Scanner sc = new Scanner(System.in);
System.out.println("Enter an digit");
int n, rem, reverce = 0;
n = sc.nextInt();
while (n > 0) {
rem = n % 10;
reverce = reverce * 10 + rem;
n = n / 10;
System.out.println(reverce);
}
}
}
