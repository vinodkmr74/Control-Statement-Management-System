import java.util.*;
public class Factorial {
public static void main(String[] args) {
Scanner sc = new Scanner(System.in);
System.out.println("Enter a factorial number");
int i, n, fac = 1;
n = sc.nextInt();
for (i = 1; i <= n; i++) {
fac = fac * i;
System.out.println("factorial" + " " + fac);
}
}
}
