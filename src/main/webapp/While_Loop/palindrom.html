import java.util.*;
public class palindrom
{ public static void main(String[] args)
{
Scanner sc = new Scanner(System.in);
System.out.println("Enter a number");
int n, rem = 0, temp;
n = sc.nextInt();
temp = n;
while (n > 0)
{
rem = (rem * 10) + n % 10;
n = n / 10;
}
if (temp == rem)
{
System.out.println("palindrom");
} else
{
System.out.println("not palindrom");
}
}
}
