import java.util.*;
public class basic_salary {
public static void main(String[] args) {
Scanner sc = new Scanner(System.in);
System.out.println("Enter Basic salary");
int bs;
bs = sc.nextInt();
int da, ta;
if (bs >= 10000) {
da = bs * 40 / 100;
ta = bs * 30 / 100;
} else {
da = bs * 30 / 100;
ta = bs * 20 / 100;
}
System.out.println("besic salary " + bs);
System.out.println("bs=" + da + " ta= " + ta);
System.out.println("Total Grow salary" + (bs + da + ta));
}
}
